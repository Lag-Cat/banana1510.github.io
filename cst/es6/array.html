# Array.from

Array.from 用于转换为真数组，在js中对象可以实现一个伪数组，伪数组可以像数组那样取值，但是其原型是 Object（对象） ，所以它不能 正常 调用 Array 的方法，但是伪数组可以通过调用Array的原型中的函数去使用Array的函数。

var a = {
    '0': 'a',
    '1': 'b',
    length: 2
}

a.find(item => {
    return item === 'd'
});
//Uncaught TypeError: a.find is not a function

Array.prototype.map.call(a, item => {
    console.log(item)
});
//a
//b

我们可以通过Array.from来将原来的伪数组转换成真数组:

var arr = {
    '0': 'a',
    '1': 'b',
    length: 2
}

// ES5的写法
var arr1 = [].slice.call(arr);

// ES6的写法
let arr2 = Array.from(arr);

# 将字符串转换为数组:

var arr = Array.from('banana');
console.log(arr);
//["b", "a", "n", "a", "n", "a"]

# 将Set转为数组

var set1 = new Set(["b", "a", "n", "a", "n", "a"]); //set会去除重复的元素
var arr = Array.from(set1);
console.log(arr);
//["b", "a", "n"]

这里可以通过Set来对数组进行去重

# 将Map转为数组

const map = new Map([
    [1, 2],
    [2, 4],
    [4, 8]
]);
Array.from(map);
// [[1, 2], [2, 4], [4, 8]]

const mapper = new Map([
    ['1', 'a'],
    ['2', 'b']
]);
Array.from(mapper.values());
// ['a', 'b'];

Array.from(mapper.keys());
// ['1', '2'];

# Array.map

Array.map的作用是遍历数组，并且返回一个新的数组：

var arr=[1,2,3,4,5,6];
var arr1=arr.map(item=>{return item*item});
console.log(arr,arr1)
// [1, 2, 3, 4, 5, 6]  [1, 4, 9, 16, 25, 36]

# Array.foreach

Array.foreach的作用是遍历数组，与Array.map不同的是，foreach不会返回新数组。

var arr=[1,2,3,4,5,6];
var arr1=arr.map((item,index,arr)=>{return item*item});
console.log(arr,arr1)
//[1,2,3,4,5,6] undefined

注意: forEach() 对于空数组是不会执行回调函数的。

# Array.filter

Array.filter的作用是用来过滤元素:

var arr = [1,2,3,4,5,6,7];
var newArr=arr.filter(
  item=>{
    return item>3;
  }
);
console.log(newArr);

# Array.find

Array.find用来查找元素，如果有多个符合的话只会返回匹配到的第一个元素

var arr=[1,2,3,4,5,6,7];
var tar=arr.find(
  item=>{ 
    return item===3;
  }
)
console.log(tar);

var arr=[{id:1,name:'小红'},{id:3,name:'小明'},{id:3,name:'小花'},{id:4,name:'小林'}];
var tar=arr.find(item=>{
    return item.id===3;
})
console.log(tar);
//{id: 3, name: "小明"}
