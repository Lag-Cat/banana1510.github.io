# 箭头函数

# 基本用法

es6中增加了箭头函数，使得我们在代码阅读上面有了更好的帮助。

以下为箭头函数基本用法：

//不带参数
let func1 = () => {
    console.log('Arrow Function!');
}

func1();
//Arrow Function!

//带参数
let func2 = (value) => {
    console.log(value);
}

func2('Arrow Function!');
//Arrow Function!

只有一个参数的时候可以简化写法:

let func1 = value => {
    console.log(value);
}
func1('Arrow Function!');
//Arrow Function!

如果函数体内只执行一行语句还可以继续简化写法:

let func1 = value => console.log(value);

func1('Arrow Function!');
//Arrow Function!

如果函数体内只有一行执行代码而且这行代码返回是一个对象，那么这个箭头函数就需要带上括号：

let func1 = value => ({
    a: value
});
console.log(func1(10))
//{a: 10}

# 与普通函数的区别

箭头函数在优化了函数的写法之余，还对普通函数中的this指向做出了修复。 在普通函数中this的指向：

//例子1
function func1() {
    console.log(this)
}
func1()
//window

//例子2
var name="小明"
function func1(){
    var name="小红"
    return this.name;
}
console.log(func1())
//小明

//例子3
var person = {
    name: "小红",
    getName: function() {
        return this.name
    }
}
var name = "小明"
console.log(person.getName()) //小红
var getName = person.getName;
console.log(getName()) //小明

在普通函数中，在下面例子中可以看出，person调用自身的getName函数时候，this的指向自身，而将person.getName赋于getName，再调用getName函数，这里相当于window调用getName，所以this指向window，由此可得普通函数中this指向的是调用它的对象。

但是再箭头函数中修复了这一点：

//例子1
let func = () => {
    console.log(this);
}
func();
//window

//例子2
var name="小明"
let func = () => {
    var name="小红";
    console.log(this.name);
}
func();
console.log(this.name);
//小明
//小明

//例子3
var person = {
    name: "小红",
    getName: () => {//定义这个对象的是window
        return this.name
    }
}
var name = "小明"
console.log(person.getName()) //小明
var getName = person.getName;
console.log(getName()) //小明
console.log(person.getName.call(person))//小明


//例子4:类的形式
function Person(){
    this.name="小红";//定义这个对象的是Person类
    this.getName=()=>{ return this.name}
}
var name="小明"
var person=new Person();
console.log(person.getName());//小红
var getName=person.getName;
console.log(getName())//小红

console.log(person.getName.call(this))//小红
console.log(person.getName.apply(this))//小红
var getName1=person.getName.bind(this)
console.log(getName1())//小红

由上可以看出，箭头函数的this指向永远指向定义它的作用域，哪怕你用的是call、apply还是bind。
