# 路由基本配置

路由配置 root/src/router/index.js

import Vue from 'vue'
import Router from 'vue-router'
import login from '@/views/login/login'
import index from '@/views/index/index'
Vue.use(Router)

const router = new Router({
    routes: [{
            path: '/',
            name: 'login',
            component: login
        },
        {
            path: '/index',
            name: 'index',
            component: index
        }
    ]
})

export default router

页面设置 App.vue

<template>
  <div id="app">
      <router-view></router-view>
  </div>
</template>
<script>
  export default {
    name: 'App',
  }
</script>

# 子路由配置

如果需要在index.vue页面中增加子路由，那么需要在login页面中加入 <router-view></router-view>

<template>
  <div>
    <h1>这里是index</h1>
    <router-view></router-view>
  </div>
</template>
<script>
  export default {
    name: 'index',
  }
</script>

并且在对应路由中添加其子项

import home from '@/views/home/home'
//...
{
    path: '/index',
    name: 'index',
    component: index,
    children: [
        path: '/index/home',
        name: 'home',
        component: home,
    ]
}

# 动态引入组件

在项目中我们通常会采用动态引入的方法来引入组件，这样能有效地降低第一次打开网页时需要加载的数据，减少请求，提高执行效率和服务器压力。 其原理是定义一个函数赋于component，此时在路由初始化的时候并不会执行函数的内容，而是在该路由加载的时候才会引入对应的组件。

//...
{
    path: '/index',
    name: 'index',
    component: index,
    children: [
        path: '/index/home',
        name: 'home',
        component: () => import('@/views/home/home'),
    ]
}

# 元数据

在开发过程中，通常我们需要添加一些信息进路由，如是否keep alive、页面的标题、是否可缓存数据等，这是就要用到元数据meta:

//...
{
    path: '/index',
    name: 'index',
    component: index,
    children: [
        path: '/index/home',
        name: 'home',
        component: () => import('@/views/home/home'),
        meta: {
            title: '个人资料',
            keepAlive: true,
            allowCache: false
        }
    ]
}

# keep-alive

keep-alive顾名思义就是保持活动状态，在keep-alive中的组件会被缓存，在前进后退的时候，keep-alive内部的组件将不会重新加载。 由此特性，我们可以实现切换页面后，一些数据可以保留在原页面中，再次访问该页面的时候不用重新发送请求数据。在这里我们利用了路由的元数据：

    <keep-alive>
        <router-view v-if="$route.meta.keepAlive" class="indexRouter"></router-view>
    </keep-alive>
    <router-view v-if="!$route.meta.keepAlive"></router-view>
