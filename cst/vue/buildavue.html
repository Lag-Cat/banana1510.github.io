# Observer

vue对数据进行劫持主要是用到get和set的方法，该方法通过Object.defineProperty或Proxy来实现。

vue2.0及2.0以下进行数据劫持使用了Object.defineProperty，在vue3.0之后改用了es6的Proxy。

vue 不支持ie8的原因是ie8中的Object.defineProperty只能对dom对象进行操作。

基于Object.defineProperty的数据劫持:

function defineReactive(obj, key, val) {
    Object.defineProperty(obj, key, {
        get: function() {
            console.log("get：", val);
            return val;
        },
        set: function(newVal) {
            val = newVal
            console.log("set:", val);
        }
    })
}
var a = {}
defineReactive(a, 'b', 1);
a.b = 3;
console.log(a.b);
//set: 3
//get： 3
//3

OK，那么我们现在了解了数据劫持的方法了，接下来我们要对我们的对象进行遍历，将对象里每一个属性都进行数据劫持。

function Observer(obj) {
    Object.keys(obj).forEach(key => {
        //如果子项仍为对象，则对对象进行遍历
        if (Object.prototype.toString.call(obj[key]) === "[Object Object]") {
            Observer(obj[key]);
        }
        defineReactive(obj, key, obj[key]);
    })
}

基于Proxy的Observer:

# Dep

vue通过Observer中数据劫持后，在Dep中进行订阅收集以及对属性变化的数据对应的订阅者进行更新。现在我们稍微改造一下Observer，以及实现Dep：

function defineReactive(obj, key, val) {
    let dep = new Dep();
    Object.defineProperty(obj, key, {
        configurable: true, //当且仅当该属性的 configurable 键值为 true 时，该属性的描述符才能够被改变，同时该属性也能从对应的对象上被删除
        enumerable: true, //当且仅当该属性的 enumerable 键值为 true 时，该属性才会出现在对象的枚举属性中。
        get: function() {
            dep.addSub(); //收集订阅者，加到dep中
            return val;
        },
        set: function(newVal) {
            val = newVal;
            dep.notify(); //对订阅者进行通知
        }
    })
}
class Dep {
    constructor() {
        this.subs = []; //订阅者列表
    }
    //订阅者
    addSub() {
        console.log("加入了订阅者")
    }
    //通知订阅者（发布者）
    notify() {
        console.log("我被更新了")
    }
}

# watcher

我们现在有了Dep收集器之后，接下来我们需要实现的是watcher，现在我们改造一下前面的方法，将watcher加进去：

class Watcher {
    constructor(viewModel, exp, callback) {
        this.cb = callback; //回调函数,用于值改变后回调绑定他的处理函数
        this.vm = viewModel; //绑定viewmodel,
        this.exp = exp; //名字
        this.value = this.get();
    }

    update() {
        this.run();
    }

    run() {
        var value = this.vm.data[this.exp];
        var oldVal = this.value;
        if (value !== oldVal) {
            this.value = value;
            this.cb.call(this.vm, value, oldVal);
        }
    }

    get() {
        Dep.target = this; // 缓存自己
        var value = this.vm.data[this.exp] // 强制执行监听器里的get函数
        Dep.target = null; // 释放自己
        return value;
    }
}

同时Observer和Dep中也要修改一下:

function defineReactive(obj, key, val) {
    let dep = new Dep();
    Object.defineProperty(obj, key, {
        configurable: true,
        enumerable: true,
        get: function() {
            if (Dep.target) //判断是否需要添加订阅者
                dep.addSub(Dep.target); //收集订阅者，加到dep中
            return val;
        },
        set: function(newVal) {
            if (val === newVal) {
                return;
            }
            val = newVal;
            dep.notify(); //对订阅者进行通知
        }
    })
}

class Dep {
    constructor() {
        this.subs = [];
        this.target = null;
    }

    //订阅者
    addSub(watcher) {
        this.subs.push(watcher)
        console.log("加入了订阅者")
    }

    //通知订阅者（发布者）
    notify() {
        this.subs.forEach(sub => {
            sub.update(); //订阅者进行更新
        })
        console.log("我被更新了")
    }
}

最后将Observer和watcher结合起来:

function MyVue(data, el, exp) {
    this.data = data;
    Observer(data);
    el.innerHTML = this.data[exp]; // 初始化模板数据的值
    new Watcher(this, exp, function(value) {
        el.innerHTML = value;
    });
    return this;
}
